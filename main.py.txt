import os
from flask import Flask, request, jsonify
import google.generativeai as genai

app = Flask(__name__)

# [철통 보안] API 키는 코드에 노출하지 않고 서버 환경변수에서만 호출
genai.configure(api_key=os.environ.get("GEMINI_API_KEY"))
model = genai.GenerativeModel('gemini-2.5-pro') # 소설 창작에 최적화된 Pro 모델

@app.route('/generate-novel', methods=['POST'])
def generate_novel():
    try:
        data = request.json
        subject = data.get('subject', '')
        
        # 전문 소설가의 문체를 위한 정교한 프롬프트 세팅
        prompt = f"당신은 당신은 노벨 문학상 후보에 오를 법한 모든 분야에 통달한 천재 현대 소설가입니다. 
    다음 주제를 바탕으로 깊이 있는 은유와 세밀한 심리 묘사가 담긴 단편 소설을 작성하세요.
      대문호입니다. 주제: '{subject}'에 대하여 독자의 심금을 울리는 문학적인 소설 한 문단을 작성하세요. [작성 규칙]
    1. 독자의 상상력을 자극하는 문학적인 문체를 사용할 것.
    2. 사건 중심이 아닌, 인물의 내면과 분위기 묘사에 집중할 것. 
    3. 결말은 여운이 남도록 구성할 것.
    4. 한국어의 아름다움을 살린 고급 어휘를 선택할 것."
        
        response = model.generate_content(prompt)
        return jsonify({"novel": response.text})
    except Exception as e:
        # 에러 발생 시에도 서버 보안을 위해 상세 로직은 숨기고 메시지만 전달
        return jsonify({"error": "창작 과정에서 일시적인 영감이 부족합니다. 다시 시도해 주세요."}), 500

if __name__ == "__main__":
    # 포트 10000번 고정 및 외부 접속 허용
    app.run(host='0.0.0.0', port=10000)
